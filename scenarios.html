🍕 Restaurant Bill Splitter

A group of friends eats at a restaurant. Write a program that:

  1. Asks for the total bill amount (Rand)
  2. Asks how many people are splitting the bill
  3. Asks if they want to add a tip (10%, 15%, or 20%)
  4. Uses a function calculate_split(total, people, tip_percent) that returns the amount per person
  5. Displays the tip amount, grand total, and per-person amount (all to 2 decimal places)
✅ Solution
def calculate_split(total, people, tip_percent):
    tip = total * (tip_percent / 100)
    grand = total + tip
    return grand / people

bill   = float(input("Total bill (R): "))
people = int(input("How many people? "))

print("Tip: 1=10%, 2=15%, 3=20%")
choice = int(input("Choose: "))

if choice == 1:
    tip_pct = 10
elif choice == 2:
    tip_pct = 15
else:
    tip_pct = 20

tip_amount = bill * (tip_pct / 100)
per_person = calculate_split(bill, people, tip_pct)

print(f"Tip: R{tip_amount:.2f}")
print(f"Grand total: R{bill + tip_amount:.2f}")
print(f"Per person: R{per_person:.2f}")
Topics tested: float/int input conversion (IPO), if/elif/else for tip selection (Decisions), function with return value (Functions), f-string formatting with :.2f (Output).

🏫 Class Mark Analyzer

A teacher needs a program that:

  1. Asks for the number of students in the class
  2. Uses a loop to enter each student's name and mark
  3. Uses a function get_grade(mark) that returns "Distinction" (75+), "Merit" (60–74), "Pass" (50–59), or "Fail" (<50)
  4. After all students, displays: the class average, how many got a Distinction, and who got the highest mark
✅ Solution
def get_grade(mark):
    if mark >= 75:   return "Distinction"
    elif mark >= 60: return "Merit"
    elif mark >= 50: return "Pass"
    else:             return "Fail"

n = int(input("Number of students: "))
total = 0
distinctions = 0
top_name = ""
top_mark = -1

for i in range(n):
    name = input(f"Student {i+1} name: ")
    mark = float(input(f"Student {i+1} mark: "))
    total += mark

    grade = get_grade(mark)
    print(f"  → {name}: {grade}")

    if grade == "Distinction":
        distinctions += 1
    if mark > top_mark:
        top_mark = mark
        top_name = name

print(f"\nClass average: {total/n:.1f}")
print(f"Distinctions: {distinctions}")
print(f"Top student: {top_name} ({top_mark})")

🌡️ Weather Data Processor

Temperature readings are stored in a file temps.txt, one number per line. Write a program that:

  1. Reads all temperatures into a list (handle the case where the file doesn't exist)
  2. Uses a function analyze_temps(temps_list) that returns a tuple: (average, highest, lowest)
  3. Displays the results to 1 decimal place
  4. Counts how many readings are above 30°C (hot days)
✅ Solution
def analyze_temps(temps):
    avg = sum(temps) / len(temps)
    return (avg, max(temps), min(temps))

try:
    temperatures = []
    with open("temps.txt", "r") as f:
        for line in f:
            temperatures.append(float(line.strip()))

    average, highest, lowest = analyze_temps(temperatures)

    print(f"Average: {average:.1f}°C")
    print(f"Highest: {highest:.1f}°C")
    print(f"Lowest:  {lowest:.1f}°C")

    hot_days = 0
    for t in temperatures:
        if t > 30:
            hot_days += 1
    print(f"Hot days (>30°C): {hot_days}")

except FileNotFoundError:
    print("temps.txt not found.")

📚 Library Book Importer

A library stores book data in a file books.txt with one book per line in the format: title,author,year. Write a program that:

  1. Creates a SQLite database library.db with a table books (id, title, author, year)
  2. Reads books.txt and inserts each book into the database
  3. Handles the case where the file doesn't exist
  4. After importing, queries and displays all books published after 2010, sorted by year
✅ Solution
import sqlite3

conn = sqlite3.connect("library.db")
cur = conn.cursor()
cur.execute("""CREATE TABLE IF NOT EXISTS books (
    id INTEGER PRIMARY KEY, title TEXT, author TEXT, year INTEGER)""")

try:
    with open("books.txt", "r") as f:
        for line in f:
            parts = line.strip().split(",")
            title  = parts[0]
            author = parts[1]
            year   = int(parts[2])
            cur.execute(
                "INSERT INTO books (title,author,year) VALUES (?,?,?)",
                (title, author, year))
    conn.commit()
    print("Import complete!")
except FileNotFoundError:
    print("books.txt not found.")

# Query recent books
cur.execute("SELECT title, author, year FROM books WHERE year > ? ORDER BY year", (2010,))
print("\nBooks after 2010:")
for title, author, year in cur.fetchall():
    print(f"  {title} by {author} ({year})")

conn.close()

🎮 Quiz Game

Create a multiple-choice quiz game that:

  1. Stores 5 questions in a list of tuples: (question, option_a, option_b, option_c, correct_answer)
  2. Loops through each question, displays the options, and asks the user for their answer (a/b/c)
  3. Uses a function check_answer(user_answer, correct) that returns True or False
  4. Tracks the score and displays the final result as a percentage
  5. At the end, tells the user if they scored "Distinction" (75%+), "Pass" (50%+), or "Fail"
✅ Solution
def check_answer(user, correct):
    return user.lower() == correct.lower()

questions = [
    ("What does CPU stand for?",
     "a) Central Processing Unit",
     "b) Central Power Unit",
     "c) Computer Personal Unit", "a"),
    ("What does 10 % 3 equal?",
     "a) 3", "b) 1", "c) 0", "b"),
    ("Which stores data permanently?",
     "a) RAM", "b) CPU", "c) Hard drive", "c"),
    ("What is len([1,2,3])?",
     "a) 2", "b) 3", "c) 4", "b"),
    ("What file mode erases content?",
     "a) r", "b) a", "c) w", "c")
]

score = 0
for i, (q, a, b, c, correct) in enumerate(questions):
    print(f"\nQ{i+1}: {q}")
    print(f"  {a}")
    print(f"  {b}")
    print(f"  {c}")
    ans = input("Your answer (a/b/c): ")
    if check_answer(ans, correct):
        print("Correct! ✓")
        score += 1
    else:
        print(f"Wrong. Answer: {correct}")

pct = score / len(questions) * 100
print(f"\nScore: {score}/{len(questions)} ({pct:.0f}%)")
if pct >= 75:   print("Distinction!")
elif pct >= 50: print("Pass")
else:              print("Fail")

📇 Contact Book

Build a contact manager that:

  1. Uses SQLite with a table contacts (id, name, phone, email)
  2. Has a function add_contact(cur, name, phone, email) that inserts a row
  3. Has a function search_contact(cur, search_name) that returns all matching rows (using LIKE for partial matches)
  4. Presents a menu loop: Add / Search / View All / Delete / Quit
  5. Handles invalid menu choices
✅ Solution
import sqlite3

def add_contact(cur, name, phone, email):
    cur.execute("INSERT INTO contacts(name,phone,email) VALUES(?,?,?)",
                (name, phone, email))

def search_contact(cur, search_name):
    cur.execute("SELECT * FROM contacts WHERE name LIKE ?",
                (f"%{search_name}%",))
    return cur.fetchall()

conn = sqlite3.connect("contacts.db")
cur = conn.cursor()
cur.execute("""CREATE TABLE IF NOT EXISTS contacts (
    id INTEGER PRIMARY KEY, name TEXT NOT NULL,
    phone TEXT, email TEXT)""")
conn.commit()

while True:
    print("\n1.Add  2.Search  3.View All  4.Delete  5.Quit")
    choice = input("Choose: ")

    if choice == "1":
        n = input("Name: ")
        p = input("Phone: ")
        e = input("Email: ")
        add_contact(cur, n, p, e)
        conn.commit()
        print("Added!")
    elif choice == "2":
        term = input("Search name: ")
        results = search_contact(cur, term)
        if results:
            for row in results:
                print(row)
        else:
            print("No matches.")
    elif choice == "3":
        cur.execute("SELECT * FROM contacts ORDER BY name")
        for row in cur.fetchall():
            print(row)
    elif choice == "4":
        did = int(input("ID to delete: "))
        cur.execute("DELETE FROM contacts WHERE id=?", (did,))
        conn.commit()
    elif choice == "5":
        break
    else:
        print("Invalid choice.")

conn.close()

🏆 Sports Club Member Manager

A sports club keeps member data in members.txt (format: name,age,sport). Write a program that:

  1. Reads the file into a list of tuples: [(name, age, sport), ...]
  2. Writes a function filter_by_sport(members, sport) that returns only members playing a specific sport
  3. Writes a function average_age(members) that returns the average age
  4. Displays: total members, average age, and all members playing "Soccer" (sorted by name)
  5. Handles missing file and invalid data (non-numeric age) gracefully
✅ Solution
def filter_by_sport(members, sport):
    result = []
    for name, age, s in members:
        if s.lower() == sport.lower():
            result.append((name, age, s))
    return result

def average_age(members):
    total = 0
    for name, age, sport in members:
        total += age
    return total / len(members)

try:
    members = []
    with open("members.txt", "r") as f:
        for line in f:
            try:
                parts = line.strip().split(",")
                members.append((parts[0], int(parts[1]), parts[2]))
            except (ValueError, IndexError):
                print(f"Skipping bad line: {line.strip()}")

    print(f"Total members: {len(members)}")
    print(f"Average age: {average_age(members):.1f}")

    soccer = filter_by_sport(members, "Soccer")
    soccer.sort()  # sorts by name (first tuple element)
    print(f"\nSoccer players ({len(soccer)}):")
    for name, age, sport in soccer:
        print(f"  {name}, age {age}")

except FileNotFoundError:
    print("members.txt not found.")

🎓 Student Records System — The Big One

Build a complete student management system that:

  1. Uses a SQLite database with a students table (id, name, module, mark)
  2. Writes a function get_grade(mark) → returns grade letter
  3. Writes a function export_results(cur, filename) → writes all student results to a text file
  4. Menu: (1) Add student (with input validation — mark must be 0–100), (2) View all, (3) Search by module, (4) Export to file, (5) Show module statistics (average, pass rate), (6) Quit
  5. All database operations must use parameterised queries and exception handling
✅ Solution
import sqlite3

def get_grade(mark):
    if mark >= 75:   return "A"
    elif mark >= 60: return "B"
    elif mark >= 50: return "C"
    else:             return "F"

def export_results(cur, filename):
    cur.execute("SELECT name, module, mark FROM students ORDER BY module, name")
    with open(filename, "w") as f:
        for name, mod, mark in cur.fetchall():
            f.write(f"{name},{mod},{mark},{get_grade(mark)}\n")

conn = sqlite3.connect("school.db")
cur = conn.cursor()
cur.execute("""CREATE TABLE IF NOT EXISTS students (
    id INTEGER PRIMARY KEY, name TEXT NOT NULL,
    module TEXT, mark REAL)""")
conn.commit()

while True:
    print("\n1.Add 2.View 3.Search 4.Export 5.Stats 6.Quit")
    ch = input("Choose: ")

    if ch == "1":
        name = input("Name: ")
        mod  = input("Module: ")
        try:
            mark = float(input("Mark (0-100): "))
            if 0 <= mark <= 100:
                cur.execute("INSERT INTO students(name,module,mark) VALUES(?,?,?)",
                            (name, mod, mark))
                conn.commit()
                print(f"Added — Grade: {get_grade(mark)}")
            else:
                print("Mark must be 0-100.")
        except ValueError:
            print("Invalid number.")

    elif ch == "2":
        cur.execute("SELECT * FROM students")
        for r in cur.fetchall(): print(r)

    elif ch == "3":
        mod = input("Module: ")
        cur.execute("SELECT name,mark FROM students WHERE module=?", (mod,))
        for n, m in cur.fetchall():
            print(f"  {n}: {m} ({get_grade(m)})")

    elif ch == "4":
        export_results(cur, "results.txt")
        print("Exported to results.txt")

    elif ch == "5":
        mod = input("Module: ")
        cur.execute("SELECT mark FROM students WHERE module=?", (mod,))
        marks = [r[0] for r in cur.fetchall()]
        if marks:
            passes = 0
            for m in marks:
                if m >= 50: passes += 1
            print(f"Average: {sum(marks)/len(marks):.1f}")
            print(f"Pass rate: {passes/len(marks)*100:.0f}%")
        else:
            print("No records found.")

    elif ch == "6":
        break

conn.close()
This scenario hits every topic: input/output, decisions (grade + validation), loops (menu + iterating results), functions (get_grade, export_results), files (export), lists (collecting marks for stats), database (full CRUD), and exceptions (try/except for validation).
